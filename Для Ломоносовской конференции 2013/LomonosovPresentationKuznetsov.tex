\section{Стохастические потоки}

\begin{frame}{Стохастические потоки}

  \begin{itemize}
  \item
    Гладкие: $dx(u,t)=\sum_{k=1}^{\infty}a_k(x(u,t))dw_k(t)$
	    $d<x(u1,t),x(u_2,t)>=b(||x(u_1,t)-x(u_2,t)||)dt$

$b$-гладкая функция. 
  \item
    Негладкие потоки
	    $d<x(u_1,t),x(u_2,t)>=b(||x(u_1,t)-x(u_2,t)||)dt$

$b$-не гладкая.
$x(,t)$ может быть разрывным. 
\item Основной вопрос: геометрические свойства $x$. 
  \end{itemize}
\end{frame}

\begin{frame}{Геометрия потоков.}

$\xi$-векторное поле на трёхмерном замкнутом односвязном многообразии $M$ с мерой $\mu$, $\xi$-бездивергентное поле на $M$. 
Рассматриваются траектории частичек, которые стартуют из точек $x_1,x_2$, под действием фазового потока поля $\xi.$
 
Асимптотический коэффициент зацепления: $\lambda_{\xi}(x_1,x_2)=\lim\limits_{T_1,T_2\to\infty}\frac{lk_{\xi}(x_1,x_2;T_1,T_2)}{T_1T_2}.$
 Имеет место такой результат: для почти всех $x_1,x_2\in M$ выполняется:
$\lambda_{\xi}(x_1,x_2)=\int\limits_{M}\int\limits_{M} \lambda_{\xi}(x_1,x_2)d\mu(x_1)d\mu(x_2).$

\begin{frame}{Вращения винеровского процесса на плоскости}

Для двумерного винеровского процесса, который выходит из точки, отличной от различных точек $A_1,\ldots,A_n$, обозначим: $\Phi_k(t)$-
``количество оборотов``, которое его траектория совершает 
вокруг точки $A_k$. Тогда $(\frac{2}{\ln t}\Phi_k(t),k=1,\ldots,n)\xrightarrow[t\to\infty]{d}(C_1,\ldots,C_n)$. 
При этом предельное распределение не зависит от выбора точек $A_1,\ldots,A_n$.

\begin{frame}{Интеграл Концевича.}
\begin{itemize}
 \item 
Интеграл Концевича
$K_m=\sum\limits_{P\in \mathbb{P}_{mn}}\int\limits_{\Delta_m}\omega_{P_{11}P_{12}}(t_1)\dots\omega_{P_{m1}P_{m2}}(t_m)D(P),$
де $\Delta_m=\Delta_m(T)=\{(t_1,\dots,t_m)\mid 0\le t_1\le\dots\le t_m \le T\},$
$\omega_{ij}(t)=\omega_{ji}(t)=\frac{1}{2\pi i}\frac{dZ_i(t)-dZ_j(t)}{Z_i(t)-Z_j(t)}.$
 \item
 Числовые инварианты: при каждом $m$ все диаграммы порядка $m$ заменяем на числа (``веса``) таким образом,
что удовлетворяются одночленное и четырёхчленное соотношения. При этой замене $K_m$ превращается в числовой инвариант. 
\item Пример: $\lambda_{kl}(t)=\int\limits_{0}^t\omega_{ij}(t')dt'.$

Интегральный инвариант второго порядка для косы из трёх струн $a,b,c$:
$2\Psi_{abc}=\int\limits_{0}^t(\lambda_{ab}-\lambda_{bc})\lambda_{ca}+(\lambda_{bc}-\lambda_{ca})\omega_{ab}+(\lambda_{ca}-\lambda_{ab})\lambda_{bc}.$
\end{itemize} 
\end{frame}

\section{Результаты}

\begin{frame}{Вычисление инвариантов Концевича для негладких траекторий.}
Интегральный инвариант Концевича для непрерывной траектории является пределом инвариантов 
для последовательности вписанных в эту траекторию ломаных, когда мелкость разбиения стремится к 0.

\begin{frame}{Основное утверждение.}
\begin{itemize}
 \item Для семимартингалов относительно общей фильтрации инварианты Концевича вычисляются как соответствующие кратные 
интегралы Стратоновича. 
 \item Пример: упомянутый выше инвариант второго порядка становится 
$\int\limits_{0}^t(\lambda_{ab}-\lambda_{bc})\circ d \lambda_{ca}+(\lambda_{bc}-
\lambda_{ca})\circ d\lambda_{ab}+(\lambda_{ca}-\lambda_{ab})\circ d\lambda_{bc}.$
\end{itemize}
\end{frame}

\begin{frame}{Идея доказательства.}
\begin{itemize}
 \item
 Имеет место такое свойство интегралов Стратоновича:
 $\sum\limits_{i=0}^{n-1}\frac{X(t_i)+X(t_{i+1})}{2}(Y(t_{i+1})-Y(t_{i}))\xrightarrow[\max \Delta t_i\to 0]{P}\int\limits_{0}^T X(t)\circ dY(t),$
 где $X, Y$ - непрерывнве семимартингалы. 
\item
 Во время доказательства непрерывная коса заменяется на ломаные, которые её приближают, а суммы 
$\sum\limits_{i=0}^{p-1}\frac{K(t_i)+K(t_{i+1})}{2}(\phi(t_{i+1})-\phi(t_i))$
для ломаных заменяются на соответствующие суммы для исходной косы. Оценивается точность замены.
\end{itemize}
\end{frame}

\item

\end{frame}
