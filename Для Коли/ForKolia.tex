\section{Неопределённые интегралы}
\begin{enumerate}
 \item
 $$\int \frac{2}{x+3}dx=2\ln|x+3|+C.$$
 \item
\begin{multline*}
\int \frac{x^3-5x+8}{x^2-2x+4}dx=\int \left(x+\frac{2x^2-9x+8}{x^2-2x+4}\right)dx=\int\left(x+2-\frac{5x}{x^2-2x+4}\right)dx=\\=
\frac{x^2}{2}+2x-\int\frac{5x}{x^2-2x+4}dx=(u=x-1)=\frac{x^2}{2}+2x-\int\frac{5u+5}{u^2+3}du=\\=
\frac{x^2}{2}+2x-\int\frac{5d(u^2+3)}{2(u^2+3)}-5\int\frac{1}{u^2+3}du+C=\\=
\frac{x^2}{2}+2x+\frac{5}{2}\ln{(u^2+3)}-\frac{5}{\sqrt{3}}\arctan{\frac{u}{\sqrt{3}}}+C=\\=
.\frac{x^2}{2}+2x+\frac{5}{2}\ln{(x^2-2x+4)}-\frac{5}{\sqrt{3}}\arctan{\frac{x-1}{\sqrt{3}}}+C.
\end{multline*}


 \item Имеем
 \begin{multline*}
\int \frac{\sin{3x}}{(1-2\cos{3x})^2}dx=\int \frac{d\cos{3x}}{-3(1-2\cos{3x})^2}dx=(u=\cos{3x})=\int \frac{du}{-3(1-2u)^2}=\\=
 \int\frac{d(1-2u)}{6(1-2u)^2}=(v=1-2u)=\int \frac{dv}{6v^2}=-\frac{1}{6v}+C=-\frac{1}{6(1-2u)}+C=\\=-\frac{1}{6(1-2\cos{3x})}+C.
 \end{multline*}

 \end{enumerate}
